\newcommand{\bcbox}[2]{%
  \begin{beamercolorbox}[wd=.5\paperwidth,ht=2.5ex,dp=1ex]{#1 in head/foot}%
    \usebeamerfont{#1 in head/foot}\hspace*{3mm}#2\hspace*{3mm}%

}

  \end{beamercolorbox}%

}
